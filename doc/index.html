Swiper Slider #back to top

In the main.js file,

                new Swiper(".feedback-items-cont", {
                  slidesPerView: 1,
                  spaceBetween: 30,
                  loop: true,
                  grabCursor: true,
                  navigation: {
                    nextEl: ".feedback-right",
                    prevEl: ".feedback-left",
                  },
                  breakpoints: {
                    // when window width is >= 1024px
                    1400: {
                      slidesPerView: 2,
                      spaceBetween: 30,
                    },
                  },
                });
